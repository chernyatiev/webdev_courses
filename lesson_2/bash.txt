;;Навигация

 pwd -  где находимся
 cd path - перейти в папку path
 
 cd .. - перейти в родительскую диреторию

 ls  - содержимое директории

 ls -all -расширенная информация

;; Изменение структуры файлов

mkdir - создать папку

touch - создать файл

mv - переместить папку

mv path1/* path2/ - переместить все файлы из папки path1 в path2

cp - копирование 

rm - удаление

;; Работа с текстом

cat

head

tail

less 

CTR+f -  на экран вперед
CTR+b - На экран назад

/sample -  поиск по шаблону

G - в конец файла
g - в наало файла

h - помощь
